\begin{equation*} \left[ \begin{array}{ccc|c} 1\amp 0\amp 0\amp b \\ 0\amp 1\amp 1\amp 0 \\ 1\amp 2\amp a\amp 3 \end{array} \right]\text{.} \end{equation*}
