\begin{align*} \det(B) \amp = b_{11}C'_{11} + b_{12}C'_{12} + \cdots + b_{1n}C'_{1n}\\ \amp = a_{11}(-C_{11}) + a_{12}(-C_{12}) + \cdots + a_{1n}(-C_{1n})\\ \amp = -(a_{11}C_{11} + a_{12}C_{12} + \cdots + a_{1n}C_{1n})\\ \amp = -\det(A)\text{.} \end{align*}
